# General parts
module				= class 
					| component	# TODO
					| container	# TODO

access				= PRIVATE
					| PROTECTED
					| PUBLIC

# OOP
class				= access TYPE name class_declaration 
					| TYPE name class_declaration

class_declaration	= block_start class_members block_end

class_members		= class_member 
					| class_member class_members

class_member		= field
					| property # TODO
					| method
					| constructor # TODO

field				= access identifier identifier ";"
					| access identifier identifier ASSIGN expression ";"

method				= access identifier identifier LPAREN args RPAREN block_start 
						  block_statement
					  block_end

block_statement		= statement
					| statement block_statement

statement			= assign_statement
					| newvar_statement
					| loop_statement
					| print_statement

newvar_statement	= identifier identifier ASSIGN expression ";"	

assign_statement	= identifier ASSIGN expression ";"		

loop_statement		= for_statement	# TODO
					| while_statement	# TODO
					| do_statement	# TODO

print_statement		= PRINT expression ";"

expression			= 

expression			= term
					| expression PLUS term
					| expression MINUS term

term				= factor
					| term STAR factor
					| term SLASH factor
					| term PERCENT factor

factor				= primary
					| MINUS factor
					| PLUS factor

primary				= IDENTIFIER
					| NUMBER
					| STRING
					| LPAREN expression RPAREN


# Lexems

#№ Static

PRIVATE				= "private"
PROTECTED			= "protected"
INTERNAL			= "internal"
PUBLIC				= "public"

TYPE				= "type"
VALTYPE				= "valtype"

ASSIGN				= "="
LPAREN				= "("
RPAREN				= ")"

PLUS				= "+"
MINUS				= "-"
STAR				= "*"
SLASH				= "/"
PERCENT				= "%"

#№ Dynamic

IDENTIFIER			= \p{L}[\w\`\$]
NUMBER				= [0-9]+(\.[0-9]+)?
STRING				= \"((?:[^\"\\]|\\.)*)\"
					| \'((?:[^\'\\]|\\.)*)\'
